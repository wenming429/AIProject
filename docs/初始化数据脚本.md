**将udmuser.EEMPLOYEEID 更新到users.userid** 

```sql

-- 在 users 表的 username 字段上创建索引
CREATE INDEX idx_users_username ON go_chat.users(username);

-- 在 udmuser 表的 LOGINNAME 字段上创建索引
CREATE INDEX idx_udmuser_loginname ON go_chat.udmuser(LOGINNAME);

-- 在 users 表的 userid 字段上创建索引
CREATE INDEX idx_users_userid ON go_chat.users(userid);

SELECT a.username, a.userid, b.LOGINNAME, b.EEMPLOYEEID
FROM go_chat.users a
INNER JOIN go_chat.udmuser b 
    -- 修改点：在关联条件后加上 COLLATE utf8mb4_general_ci
    ON a.username COLLATE utf8mb4_general_ci = b.LOGINNAME COLLATE utf8mb4_general_ci
WHERE (a.userid IS NULL OR a.userid = '');

UPDATE go_chat.users a
INNER JOIN go_chat.udmuser b 
    ON a.username COLLATE utf8mb4_general_ci = b.LOGINNAME COLLATE utf8mb4_general_ci
SET a.userid = b.EEMPLOYEEID
WHERE (a.userid IS NULL OR a.userid = '');




update users set userid =SUBSTRING_INDEX(email, '@', 1)  WHERE (userid IS NULL OR userid = '');


SELECT SUBSTRING_INDEX(u.email, '@', 1) AS username_prefix, u.userid, u.username, u.email
FROM users u 
WHERE u.userid = '';


```
