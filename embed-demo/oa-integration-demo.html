JavaScript жЎҘжҺҘ API

йҖҡиҝҮ postMessage е®һзҺ°и·ЁзӘ—еҸЈйҖҡдҝЎпјҢжҺ§еҲ¶иҒҠеӨ©зӘ—еҸЈиЎҢдёәпјҡ

/**
 * LumenIM еөҢе…ҘйӣҶжҲҗ JavaScript SDK
 * жҸҗдҫӣдёҺиҒҠеӨ©зӘ—еҸЈзҡ„йҖҡдҝЎиғҪеҠӣ
 */

class LumenIMEmbed {
  constructor(options = {}) {
    this.iframe = document.getElementById(options.iframeId || 'lumenim-chat-frame');
    this.origin = options.origin || 'https://your-im-domain.com';
    this.listeners = new Map();
    
    // зӣ‘еҗ¬жқҘиҮӘ iframe зҡ„ж¶ҲжҒҜ
    window.addEventListener('message', this.handleMessage.bind(this));
  }
  
  /**
   * еҸ‘йҖҒж¶ҲжҒҜеҲ° iframe
   */
  postMessage(type, data) {
    if (!this.iframe || !this.iframe.contentWindow) return;
    
    this.iframe.contentWindow.postMessage({
      type,
      data,
      timestamp: Date.now()
    }, this.origin);
  }
  
  /**
   * еӨ„зҗҶжқҘиҮӘ iframe зҡ„ж¶ҲжҒҜ
   */
  handleMessage(event) {
    // е®үе…Ёж ЎйӘҢпјҡйӘҢиҜҒж¶ҲжҒҜжқҘжәҗ
    if (event.origin !== this.origin) return;
    
    const { type, data } = event.data;
    
    // и§ҰеҸ‘еҜ№еә”зҡ„дәӢд»¶зӣ‘еҗ¬еҷЁ
    const callbacks = this.listeners.get(type) || [];
    callbacks.forEach(cb => cb(data));
  }
  
  /**
   * и®ўйҳ…дәӢд»¶
   */
  on(event, callback) {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, []);
    }
    this.listeners.get(event).push(callback);
  }
  
  /**
   * еҸ–ж¶Ҳи®ўйҳ…
   */
  off(event, callback) {
    const callbacks = this.listeners.get(event) || [];
    const index = callbacks.indexOf(callback);
    if (index > -1) callbacks.splice(index, 1);
  }
  
  // ========== жҺ§еҲ¶ж–№жі• ==========
  
  /**
   * жү“ејҖиҒҠеӨ©зӘ—еҸЈ
   */
  open() {
    this.postMessage('control', { action: 'open' });
  }
  
  /**
   * е…ій—ӯиҒҠеӨ©зӘ—еҸЈ
   */
  close() {
    this.postMessage('control', { action: 'close' });
  }
  
  /**
   * жңҖе°ҸеҢ–зӘ—еҸЈ
   */
  minimize() {
    this.postMessage('control', { action: 'minimize' });
  }
  
  /**
   * еҲҮжҚўеҲ°ж¶ҲжҒҜйЎөйқў
   */
  switchToMessage() {
    this.postMessage('navigate', { page: 'message' });
  }
  
  /**
   * еҲҮжҚўеҲ°йҖҡи®ҜеҪ•
   */
  switchToContact() {
    this.postMessage('navigate', { page: 'contact' });
  }
  
  /**
   * жү“ејҖжҢҮе®ҡдјҡиҜқ
   */
  openConversation(type, id) {
    this.postMessage('conversation', { type, id });
  }
  
  /**
   * и®ҫзҪ®дё»йўҳ
   */
  setTheme(theme) {
    this.postMessage('theme', { theme });
  }
}

// ========== дҪҝз”ЁзӨәдҫӢ ==========

// Go еҗҺз«Ҝ CORS й…ҚзҪ®
func corsMiddleware() gin.HandlerFunc {
  return func(c *gin.Context) {
    allowedOrigins := []string{
      "https://oa.company.com",
      "https://crm.company.com",
    }
    
    origin := c.GetHeader("Origin")
    for _, allowed := range allowedOrigins {
      if origin == allowed {
        c.Header("Access-Control-Allow-Origin", origin)
        break
      }
    }
    
    c.Header("Access-Control-Allow-Methods", "GET,POST,PUT,DELETE,OPTIONS")
    c.Header("Access-Control-Allow-Headers", "Content-Type,Authorization")
    c.Header("Access-Control-Allow-Credentials", "true")
    
    if c.Request.Method == "OPTIONS" {
      c.AbortWithStatus(204)
      return
    }
    
    c.Next()
  }
}
